魔王语言

#include<stdio.h>
#define _CRT_SECURE_NO_WARNINGS
#include <stdlib.h>
#include <string.h>
/魔王语言/

//入栈
void push(char* a, char c, int* top) {
a[top] = c;
(top)++;
}
//出栈
char pop(char a, int top) {
(top)–;
return a[top];
}
//入队
void lpush(char a, char c, int tail) {
a[tail] = c;
(tail)++;
}
//出队
char lpop(char a, int top) {
(top)++;
return a[top];
}
//判断字符为大写字母，小写字母，开闭括号中的哪一类
int iswaht(char a) {
if (a >= ‘A’ && a <= ‘Z’)
return 1;
else if (a >= ‘a’ && a <= ‘z’)
return 2;
else if (a == ‘(‘)
return 3;
else if (a == ‘)’)
return 4;
}
//判断字符为大写A还是B
int isab(char a) {
if (a == ‘A’)
return 1;
else if (a == ‘B’)
return 2;
else
return 0;
}
//将‘A’转换为‘sae’
void transformA(char stack, int stop) {
push(stack, ‘e’, stop);
push(stack, ‘a’, stop);
push(stack, ‘s’, stop);
}
//将‘B’转换为‘tAdA’
void transformB(char* stack, int* stop) {
push(stack, ‘A’, stop);
push(stack, ‘d’, stop);
push(stack, ‘A’, stop);
push(stack, ‘t’, stop);
}
//将（abcd）转换为abacada
void transformC(char* stack, int* stop) {
char mark = pop(stack, stop);
char d[1000];
int num;
for (int i = 0; stack[stop - 1] != ‘)’; i++) {
d[i] = pop(stack, stop);
num = i;
}
pop(stack, stop);
for (int i = 0; i <= num; i++) {
push(stack, mark, stop);
push(stack, d[i], stop);
}
push(stack, mark, stop);
}
//将对应的英文转换为英文
void transformCE(char line, int ltail) {
for (int i = 0; i < ltail; i++)
{
char c = line[i];
switch (c)
{
case ‘t’:
printf(“天”);
break;
case ‘d’:
printf(“地”);
break;
case ‘s’:
printf(“上”);
break;
case ‘a’:
printf(“一只”);
break;
case ‘e’:
printf(“鹅”);
break;
case ‘z’:
printf(“追”);
break;
case ‘g’:
printf(“赶”);
break;
case ‘x’:
printf(“下”);
break;
case ‘n’:
printf(“蛋”);
break;
case ‘h’:
printf(“恨”);
}
}
}
int main() {
char a[1000], stack[1000], line[1000];
char c;
printf(“请输入需要翻译的魔王语言：”);
scanf(“%s”, &a);
int length = 0;
int stop = 0, ltop = 0, ltail = 0;//栈顶指针，队列顶，队列尾
for (int i = 0; a[i] != ‘\0’; i++) {
length++;
}
for (int i = 0; i < length; i++) {
push(stack, a[length - i - 1], &stop);
}
//printf(“%s”, stack);
while (stop != 0) {
c = pop(stack, &stop);
//printf(“%c”, c);
if (iswaht(c) == 1) {
if (isab(c) == 1) {
transformA(stack, &stop);
}
else if (isab(c) == 2) {
transformB(stack, &stop);
}
}
else if (iswaht(c) == 2) {
lpush(line, c, &ltail);
}
else if (iswaht(c) == 3) {
transformC(stack, &stop);
}
}
int right = 1;
for (int i = 0; i < ltail; i++)
{
char c = line[i];
if (c != ‘a’ && c != ‘t’ && c != ‘d’ && c != ‘s’ && c != ‘e’ && c != ‘z’ && c != ‘g’ && c != ‘x’ && c != ‘h’ && c != ‘n’)
{
right = 0;
}
}
if (right == 1) {
printf(“魔王语言翻译后为：”);
for (int i = 0; i < ltail; i++)
{
char c = line[i];
printf(“%c”, c);
}
printf(“\n”);
transformCE(line, ltail);
}
else {
printf(“你tm说nm呢？”);
}
return 0;
}
