约瑟夫环

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
typedef struct huan {
int number;
int password;
struct huan* next;
}LinkList, * LLinkList;

LLinkList creat() {
LLinkList huan = (LLinkList)malloc(sizeof(LinkList));
huan->next = NULL;
return huan;
}
LLinkList add(int x, int y, LLinkList huan) {
LLinkList p = (LLinkList)malloc(sizeof(LinkList));
p->number = y;
p->password = x;
p->next = huan;
return p;
}
void print(int n, int m, LLinkList huan) {
LLinkList p = huan;
for (int j = 0; j < n; j++) {
for (int i = 0; i < m - 1; i++) {
p = p->next;
}
printf(“第%d个出位人的编号是:%d\n”, j + 1, p->next->number);
m = p->next->password;
p->next = p->next->next;
}
}
int main() {
int n;
int m;
printf(“请输入人数：”);
scanf(“%d”, &n);
LLinkList huan = creat();
LLinkList p = huan;
for (int i = 0; i < n; i++) {
int x[1000];
printf(“请输入第%d个人密码：”, i + 1);
scanf(“%d”, &x[i]);
p->next = add(x[i], i + 1, huan->next);
p = p->next;
}
printf(“请输入初始密码：”);
scanf(“%d”, &m);
print(n, m, huan);
return 0;
}
